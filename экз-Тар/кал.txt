        const roleSelect = document.getElementById('role');
        const adminNav = document.getElementById('adminNav');
        const teacherNav = document.getElementById('teacherNav');
        const studentNav = document.getElementById('studentNav');
        const userName = document.getElementById('userName');
        const userRole = document.getElementById('userRole');
        const userInitials = document.getElementById('userInitials');
        const logoutBtn = document.getElementById('logoutBtn');
        const pageTitle = document.getElementById('pageTitle');
        const toggleSidebar = document.getElementById('toggleSidebar');
        const sidebar = document.querySelector('.sidebar');
        const content = document.querySelector('.content');
        
        // Tab navigation
        const navItems = document.querySelectorAll('.nav-item');
        const tabContents = document.querySelectorAll('.tab-content');
        
        // Sample data
        const users = [
            { name: "Admin User", email: "admin@school.com", role: "Administrator", initials: "AD" },
            { name: "John Teacher", email: "john@school.com", role: "Teacher", initials: "JT" },
            { name: "Alice Student", email: "alice@school.com", role: "Student", initials: "AS" }
        ];
        
        // Initialize
        function init() {
            // Show login screen by default
            loginScreen.classList.remove('hidden');
            adminContent.classList.add('hidden');
            teacherContent.classList.add('hidden');
            studentContent.classList.add('hidden');
            
            // Hide all role-specific navs
            adminNav.classList.add('hidden');
            teacherNav.classList.add('hidden');
            studentNav.classList.add('hidden');
            
            // Set up event listeners
            setupEventListeners();
        }
        
        // Set up event listeners
        function setupEventListeners() {
            // Login form submission
            loginForm.addEventListener('submit', function(e) {
                e.preventDefault();
                const email = document.getElementById('email').value;
                const password = document.getElementById('password').value;
                const role = roleSelect.value;
                
                // Simple validation
                if (!email || !password || !role) {
                    alert('Please fill in all fields');
                    return;
                }
                
                // For demo purposes, we'll just use the role to determine the view
                loginUser(role);
            });
            
            // Logout button
            logoutBtn.addEventListener('click', function() {
                // Show login screen
                loginScreen.classList.remove('hidden');
                adminContent.classList.add('hidden');
                teacherContent.classList.add('hidden');
                studentContent.classList.add('hidden');
                
                // Reset form
                loginForm.reset();
            });
            
            // Tab navigation
            navItems.forEach(item => {
                item.addEventListener('click', function() {
                    // Remove active class from all nav items
                    navItems.forEach(navItem => {
                        navItem.classList.remove('active-nav');
                    });
                    
                    // Add active class to clicked nav item
                    this.classList.add('active-nav');
                    
                    // Get the tab to show
                    const tabToShow = this.getAttribute('data-tab');
                    
                    // Update page title
                    updatePageTitle(this.querySelector('.nav-text').textContent);
                    
                    // Hide all tab contents
                    tabContents.forEach(content => {
                        content.classList.remove('active');
                    });
                    
                    // Show the selected tab content
                    document.getElementById(tabToShow).classList.add('active');
                });
            });
            
            // Toggle sidebar
            toggleSidebar.addEventListener('click', function() {
                sidebar.classList.toggle('collapsed');
                content.classList.toggle('ml-64');
                content.classList.toggle('ml-20');
                
                // Change icon and text
                const icon = this.querySelector('i');
                const text = this.querySelector('.nav-text');
                
                if (sidebar.classList.contains('collapsed')) {
                    icon.classList.remove('fa-chevron-left');
                    icon.classList.add('fa-chevron-right');
                    text.textContent = 'Expand';
                } else {
                    icon.classList.remove('fa-chevron-right');
                    icon.classList.add('fa-chevron-left');
                    text.textContent = 'Collapse';
                }
            });
        }
        
        // Login user
        function loginUser(role) {
            // Hide login screen
            loginScreen.classList.add('hidden');
            
            // Show appropriate content based on role
            if (role === 'admin') {
                adminContent.classList.remove('hidden');
                adminNav.classList.remove('hidden');
                teacherNav.classList.add('hidden');
                studentNav.classList.add('hidden');
                
                // Set user info
                const adminUser = users.find(u => u.role === 'Administrator');
                updateUserInfo(adminUser);
                
                // Set first nav item as active
                document.querySelector('.nav-item').classList.add('active-nav');
                updatePageTitle('Users Management');
            } else if (role === 'teacher') {
                teacherContent.classList.remove('hidden');
                adminNav.classList.add('hidden');
                teacherNav.classList.remove('hidden');
                studentNav.classList.add('hidden');
                
                // Set user info
                const teacherUser = users.find(u => u.role === 'Teacher');
                updateUserInfo(teacherUser);
                
                // Set first nav item as active
                document.querySelector('#teacherNav .nav-item').classList.add('active-nav');
                updatePageTitle('My Clubs');
            } else if (role === 'student') {
                studentContent.classList.remove('hidden');
                adminNav.classList.add('hidden');
                teacherNav.classList.add('hidden');
                studentNav.classList.remove('hidden');
                
                // Set user info
                const studentUser = users.find(u => u.role === 'Student');
                updateUserInfo(studentUser);
                
                // Set first nav item as active
                document.querySelector('#studentNav .nav-item').classList.add('active-nav');
